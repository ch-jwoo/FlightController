445__INLINE static uint32_t getOscClk (uint32_t xtal, uint32_t oscSrc) {



446  uint32_t oscClk = XTALI;



447



448  switch (oscSrc) {                      /* switch OSCSRC */



449    case 0:                              /* MOSC Main oscillator */



450      switch (xtal) {                    /* switch XTAL */



451        case 0x0:



452          oscClk = 1000000UL;



453          break;



454        case 0x1:



455          oscClk = 1843200UL;



456          break;



457        case 0x2:



458          oscClk = 2000000UL;



459          break;



460        case 0x3:



461          oscClk = 2457600UL;



462          break;



463        case 0x4:



464          oscClk = 3579545UL;



465          break;



466        case 0x5:



467          oscClk = 3686400UL;



468          break;



469        case 0x6:



470          oscClk = 4000000UL;



471          break;



472        case 0x7:



473          oscClk = 4096000UL;



474          break;



475        case 0x8:



476          oscClk = 4915200UL;



477          break;



478        case 0x9:



479          oscClk = 5000000UL;



480          break;



481        case 0xA:



482          oscClk = 5120000UL;



483          break;



484        case 0xB:



485          oscClk = 6000000UL;



486          break;



487        case 0xC:



488          oscClk = 6144000UL;



489          break;



490        case 0xD:



491          oscClk = 7372800UL;



492          break;



493        case 0xE:



494          oscClk = 8000000UL;



495          break;



496        case 0xF:



497          oscClk = 8192000UL;



498          break;



499        case 0x10:



500          oscClk = 10000000UL;



501          break;



502        case 0x11:



503          oscClk = 12000000UL;



504          break;



505        case 0x12:



506          oscClk = 12288000UL;



507          break;



508        case 0x13:



509          oscClk = 13560000UL;



510          break;



511        case 0x14:



512          oscClk = 14318180UL;



513          break;



514        case 0x15:



515          oscClk = 16000000UL;



516          break;



517        case 0x16:



518          oscClk = 16384000UL;



519          break;



520       }



521      break;



522    case 1:                         /* IOSC Internal oscillator */



523      oscClk = XTALI;



524      break;



525    case 2:                         /* IOSC/4 Internal oscillator/4 */



526      oscClk = XTALI/4;



527      break;



528    case 3:                         /* 30kHz internal oscillator  */



529      oscClk = XTAL30K;



530      break;



531  }



532



533  return oscClk;



534}

536void SystemCoreClockUpdate (void)            /* Get Core Clock Frequency      */



537{



538    uint32_t rcc, rcc2;



539



540    /* Determine clock frequency according to clock register values */



541    rcc  = SYSCTL->RCC;



542    rcc2 = SYSCTL->RCC; //SYSCTL->RCC2;



543



544  //if (rcc2 & SYSCTL_RCC2_USERCC2)



545    if (rcc2 & (1UL<<31)) {                             /* is rcc2 is used ? */



546  //  if (rcc2 & SYSCTL_RCC2_BYPASS2)



547      if (rcc2 & (1UL<<11)) {                           /* check BYPASS */



548        SystemCoreClock = getOscClk (((rcc>>6) & 0x0F),((rcc2>>4) & 0x07));



549      } else {



550        SystemCoreClock = PLL_CLK;



551      }



552      if (rcc & (1UL<<22)) {                            /* check USESYSDIV */



553        if (rcc2 & (1UL<<11)) {



554          SystemCoreClock = SystemCoreClock / (((rcc2>>23) & (0x3F)) + 1);



555        } else {



556          SystemCoreClock = SystemCoreClock / (((rcc2>>23) & (0x3F)) + 1) / 2;



557        }



558      }



559    } else {



560  //    if (RCC_Val & (1UL<<11)) {                            /* check BYPASS */



561      if (rcc & (1UL<<11)) {                            /* check BYPASS */ /* Simulation does not work at this point */



562        SystemCoreClock = getOscClk (((rcc>>6) & 0x0F),((rcc>>4) & 0x03));



563      } else {



564        SystemCoreClock = PLL_CLK;



565      }



566  //  if (rcc & SYSCTL_RCC_USE_SYSDIV)



567      if (rcc & (1UL<<22)) {                            /* check USESYSDIV */



568  //    if (rcc2 & SYSCTL_RCC_BYPASS)

570  //      if (RCC_Val & (1UL<<11)) {                          /* check BYPASS */

572        } else {



573          SystemCoreClock = SystemCoreClock / (((rcc>>23) & (0x0F)) + 1) / 2;



574        }



575      }



576    }



577}
